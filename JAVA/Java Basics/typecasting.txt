
public class Literals {
	/*literals are the constant 
	values that appear directly in the program. 
	It can be assigned directly to a variable.*/
	
	public static void main(String args[]) {
		int bookPrize = 300 ; //numerical literals
		boolean isRight = true ; //boolean literals 
		char ch = 'c'; //Character literals
		int octalNumber = 012 ; //any number with a starting of 0 is called octal number
		long lng = 8L ; // put "L" in the end of the number id called long literal
		System.out.println(ch);
		System.out.println("Hi Chith");//anyone is in double  called String literals 
		System.out.println(octalNumber);
	}
}
